üíª Quick Start

Installation

# npm npm install @smart-linkify/core # yarn yarn add @smart-linkify/core # pnpm pnpm add @smart-linkify/core

JavaScript/TypeScript

import { linkify, presets } from '@smart-linkify/core'; // Basic usage const html = linkify("Visit https://example.com"); // With options const custom = linkify(text, { color: '#0066cc', detectEmails: true, maxLength: 30 }); // Using presets const social = linkify(text, presets.social);

React

import { SmartLinkify } from '@smart-linkify/react'; import { presets } from '@smart-linkify/core'; function App() { return ( <SmartLinkify text="Check https://example.com" options={presets.social} /> ); }
